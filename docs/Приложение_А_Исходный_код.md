# ПРИЛОЖЕНИЕ А. ЛИСТИНГ ИСХОДНОГО КОДА КЛЮЧЕВЫХ КОМПОНЕНТОВ

В данном приложении приводится исходный код ключевых функциональных модулей программного комплекса **CYBER_OPS://CORE**. Исходный код снабжен комментариями, раскрывающими логику работы алгоритмов, структуру используемых библиотек и особенности реализации.

---

## А.1. Серверная часть (Express API Server и SQLite DB)

Ниже представлен код файла `backend/server.js`, отвечающего за инициализацию веб-сервера Express.js, развертывание реляционной базы данных SQLite, регистрацию схем таблиц, проверку авторизационных JWT-токенов агентов и маршруты обработки игровых транзакций (XP, кредиты, разблокировка навыков).

```javascript
/**
 * CYBER_OPS://CORE - API SERVER Mainframe
 * Файл: backend/server.js
 * Назначение: Инициализация сервера, схемы БД, обработка API-запросов и авторизации
 */

const express = require('express');
const cors = require('cors');
const sqlite3 = require('better-sqlite3');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
require('dotenv').config();

const app = express();
const PORT = process.env.PORT || 4000;
const JWT_SECRET = process.env.JWT_SECRET || 'cyber_ops_secret_key_2026';

// ─── Middleware ───────────────────────────────────────────────────
app.use(cors());
app.use(express.json());

// ─── Database Setup (SQLite & WAL mode for high performance) ──────
const db = new sqlite3('cyber_ops.db');
db.pragma('journal_mode = WAL');
db.pragma('foreign_keys = ON');

// Инициализация структуры таблиц базы данных
db.exec(`
  -- Таблица Агентов (пользователей)
  CREATE TABLE IF NOT EXISTS agents (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    alignment TEXT DEFAULT 'UNASSIGNED', -- Фракция (White Hat, Black Hat)
    level INTEGER DEFAULT 1,
    xp INTEGER DEFAULT 0,
    next_level_xp INTEGER DEFAULT 1000,
    title TEXT DEFAULT 'RECRUIT',
    game_credits INTEGER DEFAULT 500,
    premium_credits INTEGER DEFAULT 0,
    skill_points INTEGER DEFAULT 3,
    pvp_enabled INTEGER DEFAULT 0,
    pvp_elo INTEGER DEFAULT 1000,
    pvp_wins INTEGER DEFAULT 0,
    pvp_losses INTEGER DEFAULT 0,
    pvp_streak INTEGER DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  );

  -- Разблокированные навыки (Дерево навыков)
  CREATE TABLE IF NOT EXISTS unlocked_skills (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    agent_id INTEGER NOT NULL,
    skill_id TEXT NOT NULL,
    unlocked_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (agent_id) REFERENCES agents(id),
    UNIQUE(agent_id, skill_id)
  );

  -- Лог получения очков опыта (XP)
  CREATE TABLE IF NOT EXISTS xp_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    agent_id INTEGER NOT NULL,
    source TEXT NOT NULL,          -- Источник опыта (Миссия, Книга, PvP)
    label TEXT NOT NULL,           -- Описание (Название миссии)
    xp_amount INTEGER NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (agent_id) REFERENCES agents(id)
  );

  -- Инвентарь агента (Снаряжение)
  CREATE TABLE IF NOT EXISTS inventory (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    agent_id INTEGER NOT NULL,
    item_name TEXT NOT NULL,
    item_type TEXT NOT NULL,       -- cpu, firewall, neurolink, network_adapter
    rarity TEXT DEFAULT 'common',  -- common, rare, epic, legendary
    buff TEXT,                     -- Эффект (например, "+15% к скорости взлома")
    equipped INTEGER DEFAULT 0,    -- Экипирован ли предмет
    FOREIGN KEY (agent_id) REFERENCES agents(id)
  );

  -- Сообщения во внутриигровом чате (Синдикат и Фракции)
  CREATE TABLE IF NOT EXISTS chat_messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    agent_id INTEGER NOT NULL,
    channel TEXT DEFAULT 'global', -- global, faction, syndicate
    message TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (agent_id) REFERENCES agents(id)
  );
`);

// ─── Middleware защиты роутов (Handshake JWT) ─────────────────────
function authMiddleware(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'Access denied. No token provided.' });
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    req.agentId = decoded.id;
    req.agentUsername = decoded.username;
    next();
  } catch {
    res.status(401).json({ error: 'Invalid or expired token.' });
  }
}

// ─── Алгоритм начисления опыта (XP) и Расчета Уровня ───────────────
app.post('/api/xp/add', authMiddleware, (req, res) => {
  const { amount, source, label } = req.body;
  if (!amount || amount <= 0) return res.status(400).json({ error: 'Invalid XP amount.' });

  const agent = db.prepare('SELECT * FROM agents WHERE id = ?').get(req.agentId);
  let xp = agent.xp + amount;
  let level = agent.level;
  let nextLevelXp = agent.next_level_xp;
  let bonusSP = 0;

  // Динамический пересчет уровней при переполнении лимита XP
  while (xp >= nextLevelXp) {
    level += 1;
    xp -= nextLevelXp;
    bonusSP += 2; // +2 очка навыков (Skill Points) за каждый уровень
    nextLevelXp = Math.floor(nextLevelXp * 1.15); // Прогрессивное увеличение сложности на 15%
  }

  const title = getTitleForLevel(level);
  db.prepare('UPDATE agents SET xp = ?, level = ?, next_level_xp = ?, title = ?, skill_points = skill_points + ? WHERE id = ?')
    .run(xp, level, nextLevelXp, title, bonusSP, req.agentId);

  // Запись транзакции в лог для истории активности
  db.prepare('INSERT INTO xp_log (agent_id, source, label, xp_amount) VALUES (?, ?, ?, ?)')
    .run(req.agentId, source || 'mission', label || 'Unknown', amount);

  const updatedAgent = db.prepare('SELECT skill_points FROM agents WHERE id = ?').get(req.agentId);
  res.json({ xp, level, nextLevelXp, title, skillPoints: updatedAgent.skill_points });
});

// Вспомогательная функция определения звания агента
function getTitleForLevel(level) {
  if (level >= 20) return 'ELITE OPERATIVE';
  if (level >= 15) return 'SPECIALIST';
  if (level >= 10) return 'FIELD AGENT';
  if (level >= 5) return 'OPERATIVE';
  if (level >= 3) return 'INITIATE';
  return 'RECRUIT';
}

// ─── Запуск Mainframe-сервера ─────────────────────────────────────
app.listen(PORT, () => {
  console.log(`[OK] Server listening on port ${PORT}`);
});
```

## А.2. Клиентское глобальное состояние (Zustand State Store)

Ниже приведен код файла `frontend/src/store/useStore.js`, реализующего единое реактивное хранилище состояния приложения на стороне клиента. Хранилище управляет профилем пользователя, состоянием авторизации, сессионными JWT-токенами, процессом разблокировки дерева навыков, логикой начисления опыта, а также обрабатывает сетевые ошибки (например, сбой соединения с API-сервером или таймаут).

```javascript
/**
 * CYBER_OPS://CORE - CLIENT STATE STORE
 * Файл: frontend/src/store/useStore.js
 * Библиотека: Zustand
 * Назначение: Реактивное управление состоянием, кэширование сессии, обработка ошибок связи с сервером
 */

import { create } from 'zustand';

const API_URL = 'http://localhost:4000';

export const useStore = create((set, get) => ({
  isLoggedIn: false,
  language: 'EN',
  streamerMode: false,
  token: null,
  loadError: null, // Код сетевой ошибки для вывода на экран блокировки
  agentInfo: {
    username: "UNKNOWN",
    alignment: "UNASSIGNED",
    level: 1,
    title: "RECRUIT",
    xp: 0,
    gameCredits: 500,
    premiumCredits: 0,
    nextLevelXp: 1000,
  },
  missions: [],
  inventory: [],
  skillPoints: 3,
  unlockedSkills: [],
  pvpEnabled: false,
  pvpStats: { elo: 1000, wins: 0, losses: 0, streak: 0 },
  xpLog: [],

  // Метод синхронизации локального состояния с данными от API
  setAgentData: (agent) => set({
    isLoggedIn: true,
    loadError: null,
    agentInfo: {
      username: agent.username || 'UNKNOWN',
      alignment: agent.alignment || 'UNASSIGNED',
      level: agent.level || 1,
      title: agent.title || 'RECRUIT',
      xp: agent.xp || 0,
      gameCredits: agent.gameCredits ?? 500,
      premiumCredits: agent.premiumCredits ?? 0,
      nextLevelXp: agent.nextLevelXp || 1000,
    },
    skillPoints: agent.skillPoints ?? 3,
    unlockedSkills: agent.unlockedSkills || [],
    xpLog: agent.xpLog || [],
    pvpStats: {
      elo: agent.pvpElo || 1000,
      wins: agent.pvpWins || 0,
      losses: agent.pvpLosses || 0,
      streak: agent.pvpStreak || 0,
    },
    pvpEnabled: agent.pvpEnabled === 1 || agent.pvpEnabled === true,
  }),

  // Запрос профиля агента с защитой от зависания сетевого запроса
  loadAgentProfile: async () => {
    const token = localStorage.getItem('token');
    if (!token) return;

    try {
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 8000); // Таймаут прерывания - 8 секунд

      const response = await fetch(`${API_URL}/api/agent/me`, {
        headers: {
          'Authorization': `Bearer ${token}`,
        },
        signal: controller.signal,
      });
      clearTimeout(timeoutId);

      if (!response.ok) {
        if (response.status === 401) {
          // Токен невалиден или просрочен - очищаем сессию
          localStorage.removeItem('token');
          set({ isLoggedIn: false, token: null, loadError: null });
          return;
        }
        throw new Error(`Server error: ${response.status}`);
      }

      const data = await response.json();
      get().setAgentData(data);
      set({ token });
    } catch (error) {
      console.error('Failed to load agent profile:', error);

      if (error.name === 'AbortError') {
        // Ошибка прерывания по таймауту
        set({ loadError: 'CONNECTION_TIMEOUT' });
      } else if (error.message?.includes('Failed to fetch') || error.message?.includes('NetworkError')) {
        // Офлайн статус сервера (Express-процесс упал или не запущен)
        set({ loadError: 'BACKEND_OFFLINE' });
      } else {
        // Прочие критические ошибки аутентификации
        localStorage.removeItem('token');
        set({ isLoggedIn: false, token: null, loadError: 'AUTH_FAILED' });
      }
    }
  },

  // Выход из учетной записи и обнуление стейта
  logout: () => {
    localStorage.removeItem('token');
    set({
      isLoggedIn: false,
      token: null,
      loadError: null,
      agentInfo: {
        username: "UNKNOWN",
        alignment: "UNASSIGNED",
        level: 1,
        title: "RECRUIT",
        xp: 0,
        gameCredits: 500,
        premiumCredits: 0,
        nextLevelXp: 1000,
      },
      skillPoints: 3,
      unlockedSkills: [],
      xpLog: [],
    });
  },

  // Метод отправки запроса на разблокировку навыка
  unlockSkill: async (skillId, cost = 1) => {
    const token = get().token;
    if (!token) return;

    if (get().skillPoints < cost || get().unlockedSkills.includes(skillId)) return;

    try {
      const response = await fetch(`${API_URL}/api/skills/unlock`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ skillId, cost }),
      });

      if (!response.ok) throw new Error('Failed to unlock skill');

      const data = await response.json();
      set({
        skillPoints: data.skillPoints,
        unlockedSkills: data.unlockedSkills.map(s => s.skill_id || s),
      });
    } catch (error) {
      console.error('Failed to unlock skill:', error);
    }
  }
}));
```
